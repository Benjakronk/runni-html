function checkCollision(obj1, obj2) {
const dx = obj1.x - obj2.x;
const dy = obj1.y - obj2.y;
const distance = Math.sqrt(dx * dx + dy * dy);
const minDistance = (obj1.radius + obj2.radius);

return distance < minDistance;
}

// Alternativ: Bounding box kollisjon
function boxCollision(rect1, rect2) {
return rect1.x < rect2.x + rect2.width &&
rect1.x + rect1.width > rect2.x &&
rect1.y < rect2.y + rect2.height &&
rect1.y + rect1.height > rect2.y;
}

// Spillobjekter
class GameObject {
constructor(x, y, type) {
this.x = x;
this.y = y;
this.type = type;
this.element = this.createElement();
}

createElement() {
const el = document.createElement('div');
el.className = this.type;
el.style.left = this.x + '%';
el.style.top = this.y + '%';
return el;
}

update() {
this.element.style.left = this.x + '%';
this.element.style.top = this.y + '%';
}

remove() {
if (this.element.parentNode) {
this.element.parentNode.removeChild(this.element);
}
}
}

function startGame() {

function updatePlayer() {
if (game.keys['ArrowUp'] || game.keys['w']) {
game.player.y = Math.max(0, game.player.y - CONFIG.PLAYER_SPEED);
}
if (game.keys['ArrowDown'] || game.keys['s']) {
game.player.y = Math.min(95, game.player.y + CONFIG.PLAYER_SPEED);
}
if (game.keys['ArrowLeft'] || game.keys['a']) {
game.player.x = Math.max(0, game.player.x - CONFIG.PLAYER_SPEED);
}
if (game.keys['ArrowRight'] || game.keys['d']) {
game.player.x = Math.min(95, game.player.x + CONFIG.PLAYER_SPEED);
}

const playerEl = document.getElementById('player');
playerEl.style.left = game.player.x + '%';
playerEl.style.top = game.player.y + '%';
}

function spawnObjects() {
if (Math.random() < CONFIG.SPAWN_RATE) {
if (Math.random() < 0.7) {
spawnCollectible();
} else if (Math.random() < 0.9) {
spawnEnemy();
} else {
spawnPowerUp();
}
}
}

function checkCollisions() {
// Player vs collectibles
game.collectibles.forEach((obj, index) => {
if (checkCollision(game.player, obj)) {
obj.element.classList.add('collected');
game.score += 10;
setTimeout(() => obj.remove(), 500);
game.collectibles.splice(index, 1);
}
});

// Player vs enemies
if (!game.player.invincible) {
game.enemies.forEach((enemy, index) => {
if (checkCollision(game.player, enemy)) {
game.player.health -= 20;
if (game.player.health <= 0) {
gameOver();
}
enemy.remove();
game.enemies.splice(index, 1);
}
});
}
}

function updateEnemies() {
game.enemies.forEach((enemy, index) => {
enemy.x += Math.cos(enemy.direction) * enemy.speed;
enemy.y += Math.sin(enemy.direction) * enemy.speed;

enemy.direction = Math.PI - enemy.direction;

if (enemy.y <= 0 || enemy.y >= 95) {
enemy.direction = -enemy.direction;
}

enemy.update();
});
}
